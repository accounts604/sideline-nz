function TogglePill({ label, active, onClick, testId }: { label: string; active: boolean; onClick: () => void; testId?: string }) {
return (
<button
type=“button”
onClick={onClick}
data-testid={testId}
className={cn(
“px-4 py-2.5 rounded-full border-2 text-sm font-medium transition-all”,
active
? “bg-accent text-white border-accent”
: “bg-white text-primary border-border hover:border-accent/50”
)}
>
{label}
</button>
);
}

function MultiSelectPills({ options, value, onChange, idPrefix }: { options: string[]; value: string[]; onChange: (v: string[]) => void; idPrefix: string }) {
return (
<div className="flex flex-wrap gap-2">
{options.map((opt) => {
const active = value.includes(opt);
return (
<TogglePill
key={opt}
label={opt}
active={active}
testId={`${idPrefix}-${opt.toLowerCase().replace(/\s+/g, "-")}`}
onClick={() => {
if (active) onChange(value.filter((v) => v !== opt));
else onChange([…value, opt]);
}}
/>
);
})}
</div>
);
}

function CardSelect({ items, selected, toggle }: { items: { label: string; desc: string; icon: string }[]; selected: string[]; toggle: (label: string) => void }) {
return (
<div className="grid grid-cols-1 sm:grid-cols-2 gap-3">
{items.map((it) => {
const active = selected.includes(it.label);
return (
<button
key={it.label}
type=“button”
onClick={() => toggle(it.label)}
data-testid={`need-${it.label.toLowerCase().replace(/\s+/g, "-")}`}
className={cn(
“text-left rounded-2xl border-2 p-4 transition-all”,
active ? “border-accent bg-accent text-white” : “border-border hover:border-accent/50 bg-white”
)}
>
<div className="flex items-start gap-3">
<div className="text-2xl">{it.icon}</div>
<div>
<div className="font-medium">{it.label}</div>
<div className={cn(“text-sm”, active ? “text-white/80” : “text-muted-foreground”)}>
{it.desc}
</div>
</div>
</div>
</button>
);
})}
</div>
);
}

function SelectField({ label, value, onChange, options, placeholder = “Select…” }: { label: string; value: string; onChange: (v: string) => void; options: string[]; placeholder?: string }) {
return (
<div>
<label className="block text-sm font-medium text-primary mb-2">{label}</label>
<select
className=“w-full border-2 border-border rounded-xl px-4 py-3 focus:outline-none focus:border-accent bg-white text-primary”
value={value}
onChange={(e) => onChange(e.target.value)}
>
<option value="">{placeholder}</option>
{options.map((o) => (
<option key={o} value={o}>{o}</option>
))}
</select>
</div>
);
}

                <p><span className="font-medium text-primary">Sports:</span> {form.sports.join(", ")}</p>

                <p><span className="font-medium text-primary">Needs:</span> {form.needs.join(", ")}</p>

                <p><span className="font-medium text-primary">Quantity:</span> {form.estimated_quantity}</p>

                  <p><span className="font-medium text-primary">Style:</span> {form.style_preference}</p>

                {form.fundraising_interest && (
                  <p><span className="font-medium text-primary">Fundraising:</span> {form.fundraising_interest}</p>

                {form.sponsorship_interest && (
                  <p><span className="font-medium text-primary">Sponsorship:</span> {form.sponsorship_interest}</p>

                <p><span className="font-medium text-primary">Timing:</span> {form.timing} ({form.urgency} urgency)</p>
