## 2) Create: `app/apply/team-store/page.tsx` (separate application flow)

```tsx
"use client";

import React, { useMemo, useState } from "react";

type AppData = {
  org_name: string;
  org_type: "club" | "school" | "other" | "";
  sport_codes: string[];
  region: string;

  supporter_base: string;
  promotion_commitment: string;
  approvals: string;
  revenue_use: string[];

  contact_name: string;
  contact_email: string;
  contact_phone: string;
  notes: string;
};

const SPORTS = ["Rugby", "League", "Football", "Netball", "Basketball", "Hockey", "Cricket", "Touch", "Other"];
const REVENUE_USE = ["Tour / travel", "Registrations", "Equipment", "Coaching", "Juniors development", "Community kaupapa", "Other"];

function cx(...classes: Array<string | false | undefined | null>) {
  return classes.filter(Boolean).join(" ");
}

function TogglePill({ label, active, onClick }: { label: string; active: boolean; onClick: () => void }) {
  return (
    <button
      type="button"
      onClick={onClick}
      className={cx(
        "px-3 py-2 rounded-full border text-sm transition",
        active ? "bg-black text-white border-black" : "bg-white text-black border-gray-300 hover:border-black"
      )}
    >
      {label}
    </button>
  );
}

function MultiPills({ options, value, onChange }: { options: string[]; value: string[]; onChange: (v: string[]) => void }) {
  return (
    <div className="flex flex-wrap gap-2">
      {options.map((opt) => {
        const active = value.includes(opt);
        return (
          <TogglePill
            key={opt}
            label={opt}
            active={active}
            onClick={() => (active ? onChange(value.filter((x) => x !== opt)) : onChange([...value, opt]))}
          />
        );
      })}
    </div>
  );
}

export default function ApplyTeamStorePage() {
  const totalSteps = 3;
  const [step, setStep] = useState(0);
  const [busy, setBusy] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [ok, setOk] = useState(false);

  const [data, setData] = useState<AppData>({
    org_name: "",
    org_type: "",
    sport_codes: [],
    region: "",

    supporter_base: "",
    promotion_commitment: "",
    approvals: "",
    revenue_use: [],

    contact_name: "",
    contact_email: "",
    contact_phone: "",
    notes: "",
  });

  const canNext = useMemo(() => {
    if (step === 0) return !!data.org_name && !!data.org_type && data.sport_codes.length > 0 && !!data.region;
    if (step === 1) return !!data.supporter_base && !!data.promotion_commitment && !!data.approvals && data.revenue_use.length > 0;
    if (step === 2) return !!data.contact_name && !!data.contact_phone && data.contact_email.includes("@");
    return false;
  }, [step, data]);

  async function submit() {
    setBusy(true);
    setError(null);
    try {
      const res = await fetch("/api/ghl/apply-team-store", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ ...data, offer: "team_store_partnership", submitted_at: new Date().toISOString() }),
      });
      if (!res.ok) {
        const j = await res.json().catch(() => ({}));
        throw new Error(j?.error || "Submission failed");
      }
      setOk(true);
    } catch (e: any) {
      setError(e.message || "Something went wrong");
    } finally {
      setBusy(false);
    }
  }

  if (ok) {
    return (
      <main className="max-w-2xl mx-auto px-4 py-14">
        <div className="rounded-2xl border border-neutral-200 p-8">
          <h1 className="text-2xl font-semibold">Application received ✅</h1>
          <p className="mt-2 text-neutral-600">
            If you’re a good fit, we’ll reach out to book a quick call and confirm next steps.
          </p>
        </div>
      </main>
    );
  }

  return (
    <main className="max-w-3xl mx-auto px-4 py-12">
      <div className="mb-8">
        <h1 className="text-3xl font-semibold">Apply for a Team Store</h1>
        <p className="mt-2 text-neutral-600">
          This is a selective partnership. Answer a few questions so we can confirm fit.
        </p>
      </div>

      <div className="rounded-2xl border border-neutral-200 p-6 md:p-8">
        {/* Progress */}
        <div className="mb-6">
          <div className="flex items-center justify-between text-sm mb-2">
            <div className="font-medium">Step {step + 1} of {totalSteps}</div>
            <div className="text-neutral-500">{Math.round(((step + 1) / totalSteps) * 100)}%</div>
          </div>
          <div className="h-2 bg-neutral-200 rounded-full overflow-hidden">
            <div className="h-2 bg-black" style={{ width: `${((step + 1) / totalSteps) * 100}%` }} />
          </div>
        </div>

        {/* Step 1 */}
        {step === 0 && (
          <div className="space-y-5">
            <label className="block">
              <div className="text-sm font-medium mb-1">Club / School name</div>
              <input
                className="w-full border border-neutral-300 rounded-xl px-3 py-2 focus:outline-none focus:ring-2 focus:ring-black"
                value={data.org_name}
                onChange={(e) => setData((d) => ({ ...d, org_name: e.target.value }))}
                placeholder="e.g. Ponsonby RFC"
              />
            </label>

            <label className="block">
              <div className="text-sm font-medium mb-2">Organisation type</div>
              <div className="flex flex-wrap gap-2">
                {(["club", "school", "other"] as const).map((t) => (
                  <TogglePill
                    key={t}
                    label={t === "club" ? "Club" : t === "school" ? "School" : "Other"}
                    active={data.org_type === t}
                    onClick={() => setData((d) => ({ ...d, org_type: t }))}
                  />
                ))}
              </div>
            </label>

            <div>
              <div className="text-sm font-medium mb-2">Sports / codes</div>
              <MultiPills options={SPORTS} value={data.sport_codes} onChange={(v) => setData((d) => ({ ...d, sport_codes: v }))} />
            </div>

            <label className="block">
              <div className="text-sm font-medium mb-1">Region</div>
              <input
                className="w-full border border-neutral-300 rounded-xl px-3 py-2 focus:outline-none focus:ring-2 focus:ring-black"
                value={data.region}
                onChange={(e) => setData((d) => ({ ...d, region: e.target.value }))}
                placeholder="e.g. Auckland / Waikato / BOP"
              />
            </label>
          </div>
        )}

        {/* Step 2 */}
        {step === 1 && (
          <div className="space-y-5">
            <label className="block">
              <div className="text-sm font-medium mb-1">Supporter / community size (rough)</div>
              <select
                className="w-full border border-neutral-300 rounded-xl px-3 py-2 focus:outline-none focus:ring-2 focus:ring-black"
                value={data.supporter_base}
                onChange={(e) => setData((d) => ({ ...d, supporter_base: e.target.value }))}
              >
                <option value="">Select…</option>
                <option>Under 200</option>
                <option>200–500</option>
                <option>500–1,000</option>
                <option>1,000+</option>
              </select>
            </label>

            <label className="block">
              <div className="text-sm font-medium mb-1">Can you commit to promoting store drops?</div>
              <select
                className="w-full border border-neutral-300 rounded-xl px-3 py-2 focus:outline-none focus:ring-2 focus:ring-black"
                value={data.promotion_commitment}
                onChange={(e) => setData((d) => ({ ...d, promotion_commitment: e.target.value }))}
              >
                <option value="">Select…</option>
                <option>Yes — we can post and share consistently</option>
                <option>Somewhat — limited capacity</option>
                <option>Not really — we need help with promo</option>
              </select>
            </label>

            <label className="block">
              <div className="text-sm font-medium mb-1">Approvals</div>
              <select
                className="w-full border border-neutral-300 rounded-xl px-3 py-2 focus:outline-none focus:ring-2 focus:ring-black"
                value={data.approvals}
                onChange={(e) => setData((d) => ({ ...d, approvals: e.target.value }))}
              >
                <option value="">Select…</option>
                <option>Just me</option>
                <option>Committee approval required</option>
                <option>SLT / Finance approval required</option>
                <option>Not sure yet</option>
              </select>
            </label>

            <div>
              <div className="text-sm font-medium mb-2">What would you use revenue for?</div>
              <MultiPills options={REVENUE_USE} value={data.revenue_use} onChange={(v) => setData((d) => ({ ...d, revenue_use: v }))} />
            </div>
          </div>
        )}

        {/* Step 3 */}
        {step === 2 && (
          <div className="space-y-5">
            <label className="block">
              <div className="text-sm font-medium mb-1">Contact name</div>
              <input

              disabled={busy || !canNext}
              className={cx("px-4 py-2 rounded-xl", canNext ? "bg-black text-white" : "bg-neutral-200 text-neutral-500 cursor-not-allowed")}
            >
              Next
            </button>
          ) : (
            <button
              type="button"
              onClick={submit}
              disabled={busy || !canNext}
              className={cx("px-4 py-2 rounded-xl", canNext ? "bg-black text-white" : "bg-neutral-200 text-neutral-500 cursor-not-allowed")}
            >
              {busy ? "Submitting…" : "Submit Application"}
            </button>
          )}
        </div>
      </div>
    </main>
  );
}
```

## 3) Create the API route: `app/api/ghl/apply-team-store/route.ts`

Wire to your GHL webhook (same pattern as before, but separate endpoint).

```ts
import { NextResponse } from "next/server";

export async function POST(req: Request) {
  try {
    const payload = await req.json();

    const webhookUrl = process.env.GHL_TEAM_STORE_APP_WEBHOOK_URL;
    if (!webhookUrl) {
      return NextResponse.json({ error: "Missing GHL_TEAM_STORE_APP_WEBHOOK_URL" }, { status: 500 });
    }

    const res = await fetch(webhookUrl, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(payload),
    });

    if (!res.ok) {
      const text = await res.text().catch(() => "");
      return NextResponse.json({ error: `GHL webhook error: ${res.status} ${text}` }, { status: 502 });
    }

    return NextResponse.json({ ok: true });
  } catch (e: any) {
    return NextResponse.json({ error: e.message || "Server error" }, { status: 500 });
  }
}
```
